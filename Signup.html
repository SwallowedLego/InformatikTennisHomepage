function doPost(e) { // Deine echte Sheet-ID hier eintragen! var sheet = SpreadsheetApp.openById('https://script.google.com/macros/s/AKfycbyez0rfiRzS_wHgjIn9P77pI9RnhTN4H749_js3Bt7JoEHhYzVcwWX_3zPK28ifbRykhg/exec').getSheetByName('Sheet1'); if (!sheet) { return ContentService.createTextOutput("Fehler: Sheet1 nicht gefunden!").setMimeType(ContentService.MimeType.TEXT); } var vorname = e.parameter.first_name || ''; var nachname = e.parameter.last_name || ''; var telefon = e.parameter.telephone || ''; var email = e.parameter.email || ''; var plaene = []; if (e.parameter.plan_basic) plaene.push(e.parameter.plan_basic); if (e.parameter.plan_family) plaene.push(e.parameter.plan_family); if (e.parameter.plan_premium) plaene.push(e.parameter.plan_premium); var plan = plaene.join(', '); sheet.appendRow([vorname, nachname, telefon, email, plan]); return ContentService.createTextOutput("Erfolg").setMimeType(ContentService.MimeType.TEXT); }
